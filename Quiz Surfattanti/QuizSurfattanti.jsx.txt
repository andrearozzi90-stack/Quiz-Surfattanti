import { useState } from "react";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";

const questions = [
  {
    text: "1. Cosa sono i surfattanti?",
    options: [
      "Sostanze che aumentano il peso dell’acqua",
      "Sostanze che aiutano a mescolare acqua e grasso",
      "Tipi di plastica"
    ],
    correct: 1,
  },
  {
    text: "2. Dove troviamo più spesso i surfattanti nella vita quotidiana?",
    options: [
      "Nei carburanti",
      "Nei detersivi e saponi",
      "Nell’aria"
    ],
    correct: 1,
  },
  {
    text: "3. Quale prodotto contiene quasi sempre surfattanti?",
    options: [
      "Acqua minerale",
      "Detersivo per piatti",
      "Carta"
    ],
    correct: 1,
  },
  {
    text: "4. Perché i surfattanti possono contribuire all’inquinamento delle acque?",
    options: [
      "Perché evaporano facilmente",

      "Perché rendono l’acqua più salata"
    ],
    correct: 1,
  },
  {
    text: "5. Cosa succede quando troppi surfattanti finiscono in fiumi o laghi?",
    options: [

      "I pesci crescono più velocemente"
    ],
    correct: 0,
  },

    text: "10. Qual è la scelta più sostenibile?",
    options: [
      "Prodotti ecologici e dosati correttamente",
      "Prodotti molto schiumosi",
      "Usare più prodotto possibile"
    ],
    correct: 0,
  },
];

export default function QuizSurfattanti() {
  const [current, setCurrent] = useState(0);
  const [score, setScore] = useState(0);
  const [finished, setFinished] = useState(false);
  const [selected, setSelected] = useState(null);
  const handleAnswer = (index) => {
    if (selected !== null) return;
    setSelected(index);
    if (index === questions[current].correct) {
      setScore(score + 1);
    }
  };


  const nextQuestion = () => {
    setSelected(null);
    const next = current + 1;
    if (next < questions.length) {
      setCurrent(next);
    } else {
      setFinished(true);
    }
  };

  const restartQuiz = () => {
    setCurrent(0);
    setScore(0);
    setFinished(false);
    setSelected(null);
  };


  return (
    <div className="min-h-screen flex items-center justify-center p-4 bg-gradient-to-br from-green-50 to-blue-50">
      <Card className="w-full max-w-xl rounded-2xl shadow-lg">
        <CardContent className="p-6 space-y-6">
          {!finished ? (
            <>
              <h1 className="text-xl font-semibold">
                {questions[current].text}
              </h1>


              <div className="space-y-3">
                {questions[current].options.map((opt, i) => {
                  let color = "outline";
                  if (selected !== null) {
                    if (i === questions[current].correct) color = "default";
                    else if (i === selected) color = "destructive";
                  }
                  return (
                    <Button
                      key={i}
                      className="w-full justify-start rounded-xl"
                      variant={color}
                      onClick={() => handleAnswer(i)}
                    >
                      {opt}
                    </Button>
                  );
                })}
              </div>


              {selected !== null && (
                <Button className="w-full mt-4" onClick={nextQuestion}>
                  Domanda successiva
                </Button>
              )}


              <p className="text-sm text-gray-500">
                Domanda {current + 1} di {questions.length}
              </p>
            </>
          ) : (
            <div className="text-center space-y-4">
              <h2 className="text-2xl font-bold">Quiz completato!</h2>
              <p className="text-lg">
                Hai risposto correttamente a <strong>{score}</strong> su <strong>{questions.length}</strong> domande.
              </p>
              <p className="text-sm text-gray-600">
                Anche piccoli gesti quotidiani aiutano a ridurre l’inquinamento 🌍
              </p>
              <Button className="mt-4" onClick={restartQuiz}>
                Ricomincia quiz
              </Button>
            </div>
          )}
        </CardContent>
      </Card>
    </div>
  );
}
